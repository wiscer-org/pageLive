  // Reusable helper (optional but nice)
  private query<T extends Element>(selector: string): T {
    const el = document.querySelector<T>(selector);
    if (!el) throw new Error(`Element not found: ${selector}`);
    return el;
  }

  ensureTitle(): HTMLHeadingElement {
    if (!this._titleEl || !this._titleEl.isConnected) {
      this._titleEl = this.query<HTMLHeadingElement>("h1.title");

      this._itemsContainer = this.query<HTMLElement>(".items");
    }
    return this._itemsContainer;
  }

  // Example usage

    this.ensureTitle().textContent = text;        // safe, always fresh
    this.ensureSubmitButton().disabled = false;   // safe, always fresh
